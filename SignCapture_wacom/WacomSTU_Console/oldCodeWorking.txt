using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;
using System.Drawing;
using wgssSTU;

namespace WacomSTU_Console
{
    internal class Program
    {
        // Tablet and capability
        private static Tablet m_tablet;
        private static ICapability m_capability;

        // Buttons
        private static System.Drawing.Rectangle btnOk;
        private static System.Drawing.Rectangle btnClear;

        // Track pen state
        private static bool isPenDown = false;

        // Store pen strokes
        private static List<IPenData> penDataList = new List<IPenData>();

        static void Main(string[] args)
        {
            Console.WriteLine("Connecting to Wacom STU tablet...");

            UsbDevices usbDevices = new UsbDevices();
            if (usbDevices.Count == 0)
            {
                Console.WriteLine("No tablet found!");
                return;
            }

            m_tablet = new Tablet();
            IErrorCode ec = m_tablet.usbConnect(usbDevices[0], true);
            if (ec.value != 0)
            {
                Console.WriteLine($"Failed to connect. Error: {ec.value}");
                return;
            }

            m_capability = m_tablet.getCapability();
            Console.WriteLine("Tablet connected successfully.");

            InitButtons();
            DrawButtons();

            // Enable pen inking
            m_tablet.setInkingMode(0x01);

            EnablePenEvents();

            Console.WriteLine("Pen data events active. Draw on the tablet or press buttons.");
            System.Threading.Thread.Sleep(-1); // Keep the console alive
        }
        private static void EnablePenEvents()
        {
            if (m_tablet == null) return;

            m_tablet.setInkingMode(0); // disable inking to get pen data

            ((ITabletEvents2_Event)m_tablet).onPenData += OnPenData;
        }
        private static void InitButtons()
        {
            int buttonWidth = 120;
            int buttonHeight = 50;
            int margin = 20;
            int y = m_capability.screenHeight - buttonHeight - margin;
            int totalWidth = (buttonWidth * 2) + margin;
            int startX = (m_capability.screenWidth - totalWidth) / 2;

            btnOk = new System.Drawing.Rectangle(startX, y, buttonWidth, buttonHeight);
            btnClear = new System.Drawing.Rectangle(startX + buttonWidth + margin, y, buttonWidth, buttonHeight);
        }

        private static void DrawButtons()
        {
            m_tablet.setInkingMode(0x00); // Disable inking

            Bitmap bitmap = new Bitmap(m_capability.screenWidth, m_capability.screenHeight);
            using (Graphics gfx = Graphics.FromImage(bitmap))
            {
                gfx.Clear(Color.White);

                // Draw existing pen strokes
                DrawStrokes(gfx);

                // Draw OK button
                gfx.FillRectangle(Brushes.LightGreen, btnOk);
                gfx.DrawRectangle(Pens.Black, btnOk);
                gfx.DrawString("OK", new Font("Arial", 16), Brushes.Black, btnOk.X + 20, btnOk.Y + 10);

                // Draw Clear button
                gfx.FillRectangle(Brushes.LightCoral, btnClear);
                gfx.DrawRectangle(Pens.Black, btnClear);
                gfx.DrawString("Clear", new Font("Arial", 16), Brushes.Black, btnClear.X + 10, btnClear.Y + 10);
            }

            WriteBitmapToTablet(bitmap);

            m_tablet.setInkingMode(0x01); // Re-enable inking
        }

        private static void DrawStrokes(Graphics gfx)
        {
            if (penDataList.Count < 2) return;

            using (Pen pen = new Pen(Color.Black, 2))
            {
                Point? prev = null;
                foreach (var pd in penDataList)
                {
                    if (pd.sw == 0)
                    {
                        prev = null; // Pen lifted
                        continue;
                    }

                    int x = pd.x * m_capability.screenWidth / m_capability.tabletMaxX;
                    int y = pd.y * m_capability.screenHeight / m_capability.tabletMaxY;
                    Point curr = new Point(x, y);

                    if (prev != null)
                        gfx.DrawLine(pen, prev.Value, curr);

                    prev = curr;
                }
            }
        }

        private static void OnPenData(IPenData penData)
        {
            int pixelX = penData.x * m_capability.screenWidth / m_capability.tabletMaxX;
            int pixelY = penData.y * m_capability.screenHeight / m_capability.tabletMaxY;
            Point p = new Point(pixelX, pixelY);

            if (penData.sw != 0 && !isPenDown)
            {
                // Pen just touched
                isPenDown = true;

                Console.WriteLine($"Pen down at x={pixelX}, y={pixelY}, pressure={penData.pressure}");

                if (btnOk.Contains(p))
                {
                    Console.WriteLine("OK button pressed!");
                    // Here you could process signature if needed
                }
                else if (btnClear.Contains(p))
                {
                    Console.WriteLine("Clear button pressed!");
                    penDataList.Clear();           // Clear stored strokes
                    ClearAndRedrawButtons();       // Clear tablet and redraw buttons
                }
            }

            if (penData.sw != 0)
            {
                // Store the stroke for drawing
                penDataList.Add(penData);
                DrawButtons(); // Redraw screen with new stroke
            }

            if (penData.sw == 0)
            {
                // Pen lifted
                isPenDown = false;
            }
        }

        private static void ClearAndRedrawButtons()
        {
            m_tablet.setInkingMode(0x00);

            Bitmap blank = new Bitmap(m_capability.screenWidth, m_capability.screenHeight);
            using (Graphics gfx = Graphics.FromImage(blank))
                gfx.Clear(Color.White);

            WriteBitmapToTablet(blank);

            DrawButtons();

            m_tablet.setInkingMode(0x01);
        }

        private static void WriteBitmapToTablet(Bitmap bitmap)
        {
            ProtocolHelper helper = new ProtocolHelper();
            ushort productId = m_tablet.getProductId();
            encodingFlag encFlag = (encodingFlag)helper.simulateEncodingFlag(productId, 0);

            encodingMode mode = (encFlag & (encodingFlag.EncodingFlag_16bit | encodingFlag.EncodingFlag_24bit)) != 0
                ? (m_tablet.supportsWrite() ? encodingMode.EncodingMode_24bit_Bulk : encodingMode.EncodingMode_24bit)
                : encodingMode.EncodingMode_1bit;

            bool useColor = (mode == encodingMode.EncodingMode_16bit_Bulk ||
                             mode == encodingMode.EncodingMode_16bit ||
                             mode == encodingMode.EncodingMode_24bit_Bulk ||
                             mode == encodingMode.EncodingMode_24bit);

            byte[] imageData = (byte[])helper.resizeAndFlatten(
                BitmapToByteArray(bitmap),
                0, 0,
                (ushort)bitmap.Width, (ushort)bitmap.Height,
                (ushort)m_capability.screenWidth, (ushort)m_capability.screenHeight,
                useColor,
                Scale.Scale_Fit,
                0, 0
            );

            m_tablet.writeImage((byte)mode, imageData);
        }

        private static byte[] BitmapToByteArray(Bitmap bmp)
        {
            using (var stream = new System.IO.MemoryStream())
            {
                bmp.Save(stream, System.Drawing.Imaging.ImageFormat.Png);
                return stream.ToArray();
            }
        }

    }
}
